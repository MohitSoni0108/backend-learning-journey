Module Wrapper Function. convert internally 

convert 
const secret = 100;

function add(a, b) {
  return a + b;
}

to 

(function (
  exports,
  require,
  module,
  __filename,
  __dirname
) {

  const secret = 100;

  function add(a, b) {
    return a + b;
  }

});

1. Read the file
        ↓
2. Wrap inside Module Wrapper Function
        ↓
3. Provide:
   - exports
   - require
   - module
   - __filename
   - __dirname
        ↓
4. Execute the file
        ↓
5. Return module.exports
        ↓
6. Cache the module
